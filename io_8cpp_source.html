150unsigned char *SurfaceToGrayscaleImage(cairo_surface_t *cairoSurface) {
151 int width, height;
152 unsigned char *result;
153 uint32_t *cairoImage, pixel;
154
155 if (cairo_image_surface_get_format(cairoSurface) != CAIRO_FORMAT_ARGB32 &&
156 cairo_image_surface_get_format(cairoSurface) != CAIRO_FORMAT_RGB24) {
157 puts("Can't convert weird image formats to grayscale.");
158 return NULL;
159 }
160
161 width = cairo_image_surface_get_width(cairoSurface);
162 height = cairo_image_surface_get_height(cairoSurface);
163
164 result = new unsigned char[width*height];
165 cairoImage = (uint32_t *)cairo_image_surface_get_data(cairoSurface);
166
167 for (int i = 0; i < height * width; i++) {
168 pixel = cairoImage[i];
169 // use "luminosity" method of grayscaling
170 result[i] = round(
171 (pixel>>16 &0xFF) *0.21 +
172 (pixel>>8 &0xFF) *0.71 +
173 (pixel &0xFF) *0.07);
174 }
175
176 return result;
177}

179cairo_surface_t *GrayscaleImageToSurface(const unsigned char *image,
180 const int width, const int height) {
181 // cairo's 8-bit type isn't BW, it's an alpha channel only, which would look a bit weird lol
182 cairo_surface_t *result = cairo_image_surface_create(CAIRO_FORMAT_RGB24, width, height);
183 uint32_t *resultData = (uint32_t *)cairo_image_surface_get_data(result);
184 // hopefully unnecessary
185 cairo_surface_flush(result);
186 for (long i = 0; i < width * height; i++) {
187 // equal r, g, and b components
188 resultData[i] = (image[i] << 16) + (image[i] << 8) + (image[i]);
189 }
190 cairo_surface_mark_dirty(result);
191 return result;

450// In the equations for pixel brightness both with motion blur enabled and disabled, we don't need
451// any constant scaling factor outside the integral because when d0=0, the brightness at the center
452// will be zero without any scaling. The scaling factor you usually see on a Normal distribution is
453// so that the Normal distribution integrates to one over the real line, making it a probability
454// distribution. But we want the /peak/ to be one, not the integral. motion blur enabled
455
466static decimal MotionBlurredPixelBrightness(const Vec2 &pixel, const GeneratedStar &generatedStar,
467 decimal t, decimal stddev) {
468 const Vec2 &p0 = generatedStar.position;
469 const Vec2 &delta = generatedStar.delta;
470 const Vec2 d0 = p0 - pixel;
471 return generatedStar.peakBrightness
472 * stddev*DECIMAL_SQRT(DECIMAL_M_PI) / (DECIMAL_SQRT(2)*delta.Magnitude())
473 * DECIMAL_EXP(DECIMAL_POW(d0.x*delta.x + d0.y*delta.y, 2) / (2*stddev*stddev*delta.MagnitudeSq())
474 - d0.MagnitudeSq() / (2*stddev*stddev))
475 * DECIMAL_ERF((t*delta.MagnitudeSq() + d0.x*delta.x + d0.y*delta.y) / (stddev*DECIMAL_SQRT(2)*delta.Magnitude()));
476}

479static decimal StaticPixelBrightness(const Vec2 &pixel, const GeneratedStar &generatedStar,
480 decimal t, decimal stddev) {
481 const Vec2 d0 = generatedStar.position - pixel;
482 return generatedStar.peakBrightness * t * DECIMAL_EXP(-d0.MagnitudeSq() / (2 * stddev * stddev));
483}

494static decimal CentroidImagingProbability(decimal mag, decimal cutoffMag) {
495 decimal brightness = MagToBrightness(mag);
496 decimal cutoffBrightness = MagToBrightness(cutoffMag);
497 decimal stddev = cutoffBrightness/DECIMAL(5.0);
498 // CDF of Normal distribution with given mean and stddev
499 return 1 - (DECIMAL(0.5) * (1 + DECIMAL_ERF((cutoffBrightness-brightness)/(stddev*DECIMAL_SQRT(2.0)))));
500}

548 // attitude at the middle of exposure time

550 // attitude 1 time unit after middle of exposure

555 // multiply up proportionally!

570 decimal ra = uniformDistribution(*rng) * 2*DECIMAL_M_PI;
571 // to be uniform around sphere. Borel-Kolmogorov paradox is calling
572 decimal de = DECIMAL_ASIN(uniformDistribution(*rng)*2 - 1);
573 decimal magnitude = magnitudeDistribution(*rng);
574
575 catalogWithFalse.push_back(CatalogStar(ra, de, magnitude, -1));
576 }

578 for (int i = 0; i < (int)catalogWithFalse.size(); i++) {
579 bool isTrueStar = i < (int)catalog.size();
580
581 const CatalogStar &catalogStar = catalogWithFalse[i];
582 Vec3 rotated = attitude.Rotate(catalogWithFalse[i].spatial);
583 if (rotated.x <= 0) {
584 continue;
585 }
586 Vec2 camCoords = camera.SpatialToCamera(rotated);

588 if (camera.InSensor(camCoords)) {

595 decimal peakBrightnessPerTime = zeroMagPeakPhotonDensity * MagToBrightness(catalogStar.magnitude);
596 decimal interestingThreshold = DECIMAL(0.05); // we don't need to check pixels that are expected to
597 // receive this many photons or fewer.
598 // inverse of the function defining the Gaussian distribution: Find out how far from the
599 // mean we'll have to go until the number of photons is less than interestingThreshold
600 decimal radius = DECIMAL_CEIL(DECIMAL_SQRT(-DECIMAL_LOG(interestingThreshold/peakBrightnessPerTime/exposureTime)*2*DECIMAL_M_PI*starSpreadStdDev*starSpreadStdDev));
601 Star star = Star(camCoords.x, camCoords.y,
602 radius, radius,
603 // important to invert magnitude here, so that centroid magnitude becomes larger for brighter stars.
604 // It's possible to make it so that the magnitude is always positive too, but allowing weirder magnitudes helps keep star-id algos honest about their assumptions on magnitude.
605 // we don't use its magnitude anywhere else in generation; peakBrightness was already calculated.
606 -catalogStar.magnitude);
607 generatedStars.push_back(GeneratedStar(star, peakBrightnessPerTime, delta));
608
609 // Now add the star to the input and expected lists.
610 // We do actually want to add false stars as well, because:
611 // a) A centroider isn't any worse because it picks up a false star that looks exactly like a normal star, so why should we exclude them from compare-centroids?
612 // b) We want to feed false stars into star-ids to evaluate their false-star resilience without running centroiding.
613
614 // Add all stars to expected, cause that's how we roll
615 expectedStars.push_back(star);
616 // and provide expected identifications for all of them
617 if (isTrueStar) {
618 expectedStarIds.push_back(StarIdentifier(expectedStars.size()-1, i));
619 }
620
621 // for input, though, add perturbation and stuff.
622 Star inputStar = star;
623 if (perturbationStddev > DECIMAL(0.0)) {
624 // clamp to within 2 standard deviations for some reason:
625 inputStar.position.x += std::max(std::min(perturbation1DDistribution(*rng), 2*perturbationStddev), -2*perturbationStddev);
626 inputStar.position.y += std::max(std::min(perturbation1DDistribution(*rng), 2*perturbationStddev), -2*perturbationStddev);
627 }
628 // If it got perturbed outside of the sensor, don't add it.
629 if (camera.InSensor(inputStar.position)
630 // and also don't add it if it's too dim.
631 && (cutoffMag >= 10000 // but always add the star if the cutoff is very high
632 || !isTrueStar // and always add the false stars
633 || std::bernoulli_distribution(CentroidImagingProbability(catalogStar.magnitude, cutoffMag))(*rng))) {
634 inputStars.push_back(inputStar);
635 if (isTrueStar) {
636 inputStarIds.push_back(StarIdentifier(inputStars.size()-1, i));
637 }
638 }
639 }
640 }
641
642 if (centroidsOnly) {
643 // we're outta here
644 return;
645 }

647 std::vector<decimal> photonsBuffer(image.width*image.height, 0);
648
649 for (const GeneratedStar &star : generatedStars) {
650 // delta will be exactly (0,0) when motion blur disabled
651 Vec2 earliestPosition = star.position - star.delta*(exposureTime/DECIMAL(2.0) + readoutTime/DECIMAL(2.0));
652 Vec2 latestPosition = star.position + star.delta*(exposureTime/DECIMAL(2.0) + readoutTime/DECIMAL(2.0));
653 int xMin = std::max(0, (int)std::min(earliestPosition.x - star.radiusX, latestPosition.x - star.radiusX));
654 int xMax = std::min(image.width-1, (int)std::max(earliestPosition.x + star.radiusX, latestPosition.x + star.radiusX));
655 int yMin = std::max(0, (int)std::min(earliestPosition.y - star.radiusX, latestPosition.y - star.radiusX));
656 int yMax = std::min(image.height-1, (int)std::max(earliestPosition.y + star.radiusX, latestPosition.y + star.radiusX));
657
658 // peak brightness is measured in photons per time unit per pixel, so if oversampling, we
659 // need to convert units to photons per time unit per sample
660 decimal oversamplingBrightnessFactor = oversamplingPerAxis*oversamplingPerAxis;
661
662 // the star.x and star.y refer to the pixel whose top left corner the star should appear at
663 // (and fractional amounts are relative to the corner). When we color a pixel, we ideally
664 // would integrate the intensity of the star over that pixel, but we can make do by sampling
665 // the intensity of the star at the /center/ of the pixel, ie, star.x+.5 and star.y+.5
666 for (int xPixel = xMin; xPixel <= xMax; xPixel++) {
667 for (int yPixel = yMin; yPixel <= yMax; yPixel++) {
668 // offset of beginning & end of readout compared to beginning & end of readout for
669 // center row
670 decimal readoutOffset = readoutTime * (yPixel - image.height/DECIMAL(2.0)) / image.height;
671 decimal tStart = -exposureTime/DECIMAL(2.0) + readoutOffset;
672 decimal tEnd = exposureTime/DECIMAL(2.0) + readoutOffset;
673
674 // loop through all samples in the current pixel
675 for (int xSample = 0; xSample < oversamplingPerAxis; xSample++) {
676 for (int ySample = 0; ySample < oversamplingPerAxis; ySample++) {
677 decimal x = xPixel + (xSample+DECIMAL(0.5))/oversamplingPerAxis;
678 decimal y = yPixel + (ySample+DECIMAL(0.5))/oversamplingPerAxis;
679
680 decimal curPhotons;
681 if (motionBlurEnabled) {
682 curPhotons =
683 (MotionBlurredPixelBrightness({x, y}, star, tEnd, starSpreadStdDev)
684 - MotionBlurredPixelBrightness({x, y}, star, tStart, starSpreadStdDev))
685 / oversamplingBrightnessFactor;
686 } else {
687 curPhotons = StaticPixelBrightness({x, y}, star, exposureTime, starSpreadStdDev)
688 / oversamplingBrightnessFactor;
689 }
690
691 assert(DECIMAL(0.0) <= curPhotons);
692
693 photonsBuffer[xPixel + yPixel*image.width] += curPhotons;
694 }
695 }
696 }
697 }
698 }
699
700 std::normal_distribution<decimal> readNoiseDist(DECIMAL(0.0), readNoiseStdDev);
701
702 // convert from photon counts to observed pixel brightnesses, applying noise and such.
703 imageData = std::vector<unsigned char>(image.width*image.height);
704 image.image = imageData.data();
705 for (int i = 0; i < image.width * image.height; i++) {
706 decimal curBrightness = 0;
707
708 // dark current (Constant)
709 curBrightness += darkCurrent;
710
711 // read noise (Gaussian)
712 curBrightness += readNoiseDist(*rng);
713
714 // shot noise (Poisson), and quantize
715 long quantizedPhotons;
716 if (shotNoise) {
717 // with GNU libstdc++, it keeps sampling from the distribution until it's within the min-max
718 // range. This is problematic if the mean is far above the max long value, because then it
719 // might have to sample many many times (and furthermore, the results won't be useful
720 // anyway)
721 decimal photons = photonsBuffer[i];
722 if (photons > DECIMAL(LONG_MAX) - DECIMAL(3.0) * DECIMAL_SQRT(LONG_MAX)) {
723 std::cout << "ERROR: One of the pixels had too many photons. Generated image would not be physically accurate, exiting." << std::endl;
724 exit(1);
725 }
726 std::poisson_distribution<long> shotNoiseDist(photonsBuffer[i]);
727 quantizedPhotons = shotNoiseDist(*rng);
728 } else {
729 quantizedPhotons = round(photonsBuffer[i]);
730 }
731 curBrightness += quantizedPhotons / saturationPhotons;
732
733 // std::clamp not introduced until C++17, so we avoid it.
734 decimal clampedBrightness = std::max(std::min(curBrightness, DECIMAL(1.0)), DECIMAL(0.0));
735 imageData[i] = floor(clampedBrightness * kMaxBrightness); // TODO: off-by-one, 256?
736 }
737}

743static Attitude RandomAttitude(std::default_random_engine* pReng) {
744 std::uniform_real_distribution<decimal> randomAngleDistribution(0, 1);
745
746 // normally the ranges of the Ra and Dec are:
747 // Dec: [-90 deg, 90 deg] --> [-pi/2 rad, pi/2 rad], where negative means south
748 // of the celestial equator and positive means north
749 // Ra: [0 deg, 360 deg] --> [0 rad, 2pi rad ]
750 // Roll: [0 rad, 2 pi rad]
751
752 decimal randomRa = 2 * DECIMAL_M_PI * randomAngleDistribution(*pReng);
753 decimal randomDec = (DECIMAL_M_PI / 2) - acos(1 - 2 * randomAngleDistribution(*pReng)); //acos returns a decimal in range [0, pi]
754 decimal randomRoll = 2 * DECIMAL_M_PI * randomAngleDistribution(*pReng);
755
756 Attitude randAttitude = Attitude(SphericalToQuaternion(randomRa, randomDec, randomRoll));
757
758 return randAttitude;
759}

953PipelineOutput Pipeline::Go(const PipelineInput &input) {
954 // Start executing the pipeline at the first stage that has both input and an algorithm. From
955 // there, execute each successive stage of the pipeline using the output of the last stage
956 // (human centipede) until there are no more stages set.
957 PipelineOutput result;
958
959 const Image *inputImage = input.InputImage();
960
961 const Stars *inputStars = input.InputStars();
962 const StarIdentifiers *inputStarIds = input.InputStarIds();
963
964 // if database is provided, that's where we get catalog from.
965 if (database) {
966 MultiDatabase multiDatabase(database.get());
967 const unsigned char *catalogBuffer = multiDatabase.SubDatabasePointer(kCatalogMagicValue);
968 if (catalogBuffer != NULL) {
969 DeserializeContext des(catalogBuffer);
970 result.catalog = DeserializeCatalog(&des, NULL, NULL);
971 } else {
972 std::cerr << "WARNING: That database does not include a catalog. Proceeding with the full catalog." << std::endl;
973 result.catalog = input.GetCatalog();
974 }
975 } else {
976 result.catalog = input.GetCatalog();
977 }
978
979 if (centroidAlgorithm && inputImage) {
980
981 std::cout << "Running centroiding algorithm..." << std::endl;
982
983 // run centroiding, keeping track of the time it takes
984 std::chrono::time_point<std::chrono::steady_clock> start = std::chrono::steady_clock::now();
985
986 // TODO: we should probably modify Go to just take an image argument
987 Stars unfilteredStars = centroidAlgorithm->Go(inputImage->image, inputImage->width, inputImage->height);
988
989 std::chrono::time_point<std::chrono::steady_clock> end = std::chrono::steady_clock::now();
990 result.centroidingTimeNs = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();
991
992 // MAGNITUDE FILTERING
993 int minMagnitude = centroidMinMagnitude;
994 if (centroidMinStars > 0
995 // don't need to filter if we don't even have that many stars
996 && centroidMinStars < (int)unfilteredStars.size()) {
997
998 Stars magSortedStars = unfilteredStars;
999 // sort descending
1000 std::sort(magSortedStars.begin(), magSortedStars.end(), [](const Star &a, const Star &b) { return a.magnitude > b.magnitude; });
1001 minMagnitude = std::max(minMagnitude, magSortedStars[centroidMinStars - 1].magnitude);
1002 }
1003 // determine the minimum magnitude according to sorted stars
1004 Stars *filteredStars = new std::vector<Star>();
1005 for (const Star &star : unfilteredStars) {
1006 assert(star.magnitude >= 0); // catalog stars can have negative magnitude, but by our
1007 // conventions, centroids shouldn't.
1008 if (star.magnitude >= minMagnitude) {
1009 filteredStars->push_back(star);
1010 }
1011 }
1012 result.stars = std::unique_ptr<Stars>(filteredStars);
1013 inputStars = filteredStars;
1014
1015 // any starid set up to this point needs to be discarded, because it's based on input
1016 // centroids instead of our new centroids.
1017 inputStarIds = NULL;
1018 result.starIds = NULL;
1019 } else if (centroidAlgorithm) {
1020 std::cerr << "ERROR: Centroid algorithm specified, but no input image to run it on." << std::endl;
1021 exit(1);
1022 } else {
1023 std::cout << "Issue found..." << std::endl;
1024 }
1025
1026 if (starIdAlgorithm && database && inputStars && input.InputCamera()) {
1027 // TODO: don't copy the vector!
1028 std::chrono::time_point<std::chrono::steady_clock> start = std::chrono::steady_clock::now();
1029
1030 result.starIds = std::unique_ptr<StarIdentifiers>(new std::vector<StarIdentifier>(
1031 starIdAlgorithm->Go(database.get(), *inputStars, result.catalog, *input.InputCamera())));
1032
1033 std::chrono::time_point<std::chrono::steady_clock> end = std::chrono::steady_clock::now();
1034 result.starIdTimeNs = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();
1035
1036 inputStarIds = result.starIds.get();
1037 } else if (starIdAlgorithm) {
1038 std::cerr << "ERROR: Star ID algorithm specified but cannot run because database, centroids, or camera are missing." << std::endl;
1039 exit(1);
1040 }
1041
1042
1043
1044 if (attitudeEstimationAlgorithm && inputStarIds && input.InputCamera()) {
1045 assert(inputStars); // ensure that starIds doesn't exist without stars
1046 std::chrono::time_point<std::chrono::steady_clock> start = std::chrono::steady_clock::now();
1047
1048 result.attitude = std::unique_ptr<Attitude>(
1049 new Attitude(attitudeEstimationAlgorithm->Go(*input.InputCamera(), *inputStars, result.catalog, *inputStarIds)));
1050
1051 std::chrono::time_point<std::chrono::steady_clock> end = std::chrono::steady_clock::now();
1052 result.attitudeEstimationTimeNs = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();
1053 } else if (attitudeEstimationAlgorithm) {
1054 std::cerr << "ERROR: Attitude estimation algorithm set, but either star IDs or camera are missing. One reason this can happen: Setting a centroid algorithm and attitude algorithm, but no star-id algorithm -- that can't work because the input star-ids won't properly correspond to the output centroids!" << std::endl;
1055 exit(1);
1056 }
1057
1058 return result;
1059}

1108static std::multimap<int, int> FindClosestCentroids(decimal threshold,
1109 const Stars &one,
1110 const Stars &two) {
1111 std::multimap<int, int> result;
1112
1113 for (int i = 0; i < (int)one.size(); i++) {
1114 std::vector<std::pair<decimal, int>> closest;
1115 for (int k = 0; k < (int)two.size(); k++) {
1116 decimal currDistance = (one[i].position - two[k].position).Magnitude();
1117 if (currDistance <= threshold) {
1118 closest.emplace_back(currDistance, k);
1119 }
1120 }
1121 std::sort(closest.begin(), closest.end());
1122 for (const std::pair<decimal, int> &pair : closest) {
1123 result.emplace(i, pair.second);
1124 }
1125 }
1126
1127 return result;
1128}

1135CentroidComparison CentroidsCompare(decimal threshold,
1136 const Stars &expected,
1137 const Stars &actual) {
1138
1139 // TODO: Somehow penalize when multiple centroids correspond to the same expected star (i.e.,
1140 // one star turned into multiple centroids). That should probably be considered an extra
1141 // centroid, but rn it isn't.
1142
1143 CentroidComparison result;
1144 // maps from indexes in each list to the closest centroid from other list
1145 std::multimap<int, int> actualToExpected = FindClosestCentroids(threshold, actual, expected);
1146
1147 for (int i = 0; i < (int)actualToExpected.size(); i++) {
1148 auto closest = actualToExpected.find(i);
1149 if (closest == actualToExpected.end()) {
1150 result.numExtraCentroids++;
1151 } else {
1152 result.meanError += (actual[i].position - expected[closest->second].position).Magnitude();
1153 result.numCorrectCentroids++;
1154 }
1155 }
1156 result.meanError /= result.numCorrectCentroids;
1157
1158 return result;
1159}

1180StarIdComparison StarIdsCompare(const StarIdentifiers &expected, const StarIdentifiers &actual,
1181 // use these to map indices to names for the respective lists of StarIdentifiers
1182 const Catalog &expectedCatalog, const Catalog &actualCatalog,
1183 decimal centroidThreshold,
1184 const Stars &expectedStars, const Stars &inputStars) {
1185
1186 StarIdComparison result = {
1187 0, // correct
1188 0, // incorrect
1189 0, // total
1190 };
1191
1192 // EXPECTED STAR IDS
1193
1194 // map from expected star indices to expected catalog indices (basically flattening the expected star-ids)
1195 std::vector<int> expectedCatalogIndices(expectedStars.size(), -1);
1196 for (const StarIdentifier &starId : expected) {
1197 assert(0 <= starId.starIndex && starId.starIndex <= (int)expectedStars.size());
1198 assert(0 <= starId.catalogIndex && starId.catalogIndex <= (int)expectedCatalog.size());
1199 expectedCatalogIndices[starId.starIndex] = starId.catalogIndex;
1200 }
1201
1202 // FIND NEAREST CENTROIDS
1203
1204 std::multimap<int, int> inputToExpectedCentroids = FindClosestCentroids(centroidThreshold, inputStars, expectedStars);
1205 // std::multimap<int, int> expectedToInputCentroids = FindClosestCentroids(centroidThreshold, expectedStars, inputStars);
1206
1207 // COMPUTE TOTAL
1208 // Count the number of expected stars with at least one input star near them
1209 for (int i = 0; i < (int)inputStars.size(); i++) {
1210 // make sure there's at least one expected star near this input star which has an identification
1211 auto closestRange = inputToExpectedCentroids.equal_range(i);
1212 bool found = false;
1213 for (auto it = closestRange.first; it != closestRange.second; it++) {
1214 if (expectedCatalogIndices[it->second] != -1) {
1215 found = true;
1216 break;
1217 }
1218 }
1219 if (found) {
1220 result.numTotal++;
1221 }
1222 }
1223
1224 // COMPUTE CORRECT AND INCORRECT
1225
1226 std::vector<bool> identifiedInputCentroids(inputStars.size(), false);
1227 for (const StarIdentifier &starId : actual) {
1228 // as later, there shouldn't be duplicate starIndex. This indicates a bug in the star-id algorithm, not comparison code.
1229 assert(!identifiedInputCentroids[starId.starIndex]);
1230 identifiedInputCentroids[starId.starIndex] = true;
1231 assert(0 <= starId.starIndex && starId.starIndex <= (int)inputStars.size());
1232 assert(0 <= starId.catalogIndex && starId.catalogIndex <= (int)actualCatalog.size());
1233
1234 // Check that there's at least one expected centroid in range which agrees with your identification.
1235 auto expectedCentroidsInRange = inputToExpectedCentroids.equal_range(starId.starIndex);
1236 bool found = false;
1237 for (auto it = expectedCentroidsInRange.first; it != expectedCentroidsInRange.second; it++) {
1238 int expectedCatalogIndex = expectedCatalogIndices[it->second];
1239 if (expectedCatalogIndex != -1
1240 && expectedCatalog[expectedCatalogIndex].name == actualCatalog[starId.catalogIndex].name) {
1241
1242 result.numCorrect++;
1243 found = true;
1244 break;
1245 }
1246 }
1247
1248 // Either there's no expected centroid in range, or none of them agree with the identification.
1249 if (!found) {
1250 result.numIncorrect++;
1251 }
1252 }
1253
1254 return result;
1255}

1592 decimal attitudeErrorSum = 0.0f;
1593 int numCorrect = 0;

1613 decimal fractionCorrect = DECIMAL(numCorrect) / expected.size();
1614 decimal fractionIncorrect = DECIMAL(numIncorrect) / expected.size();

1616 os << "attitude_error_mean " << attitudeErrorMean << std::endl;

1621static void PrintTimeStats(std::ostream &os, const std::string &prefix, const std::vector<long long> &times) {
1622 assert(times.size() > 0);
1623
1624 // print average, min, max, and 95% max
1625 long long sum = 0;
1626 long long min = LONG_MAX;
1627 long long max = 0;
1628 for (int i = 0; i < (int)times.size(); i++) {
1629 assert(times[i] > 0);
1630 sum += times[i];
1631 min = std::min(min, times[i]);
1632 max = std::max(max, times[i]);
1633 }
1634 long average = sum / times.size();
1635 std::vector<long long> sortedTimes = times;
1636 std::sort(sortedTimes.begin(), sortedTimes.end());
1637 // what really is the 95th percentile? Being conservative, we want to pick a value that at least
1638 // 95% of the times are less than. This means: (1) finding the number of times, (2) Finding
1639 // Math.ceil(0.95 * numTimes), and (3) subtracting 1 to get the index.
1640 int ninetyFiveIndex = (int)std::ceil(0.95 * times.size()) - 1;
1641 assert(ninetyFiveIndex >= 0);
1642 long long ninetyFifthPercentile = sortedTimes[ninetyFiveIndex];
1643
1644 os << prefix << "_average_ns " << average << std::endl;
1645 os << prefix << "_min_ns " << min << std::endl;
1646 os << prefix << "_max_ns " << max << std::endl;
1647 os << prefix << "_95%_ns " << ninetyFifthPercentile << std::endl;
1648}

1751 const std::vector<PipelineOutput> &actual,
1752 const PipelineOptions &values) {
1753 if (actual.size() == 0) {
1754 std::cerr << "ERROR: No output! Did you specify any input images? Try --png or --generate." << std::endl;
1755 exit(1);
1756 }
1757
1758 assert(expected.size() == actual.size() && expected.size() > 0);
1759
1760 // TODO: Remove the asserts and print out more reasonable error messages.
1761
1762#define LOST_PIPELINE_COMPARE(precondition, errmsg, comparator, path, isBinary) do { \
1763 if (precondition) { \
1764 UserSpecifiedOutputStream pos(path, isBinary); \
1765 comparator(pos.Stream(), expected, actual, values); \
1766 } else { \
1767 std::cerr << "ERROR: Comparator not applicable: " << errmsg << std::endl; \
1768 exit(1); \
1769 } \
1770 } while (0)
1771
1772 if (values.plotRawInput != "") {
1773 LOST_PIPELINE_COMPARE(expected[0]->InputImage() && expected.size() == 1,
1774 "--plot-raw-input requires exactly 1 input image, but " + std::to_string(expected.size()) + " many were provided.",
1775 PipelineComparatorPlotRawInput, values.plotRawInput, true);

1780 "--plot-input requires exactly 1 input image, and for centroids to be available on that input image. " + std::to_string(expected.size()) + " many input images were provided.",
1781 PipelineComparatorPlotInput, values.plotInput, true);

1786 PipelineComparatorPlotExpected, values.plotExpected, true);

1791 PipelineComparatorPlotOutput, values.plotOutput, true);

1796 PipelineComparatorPrintExpectedCentroids, values.printExpectedCentroids, false);
